```
    "dbaeumer.vscode-eslint",
    "esbenp.prettier-vscode",
    "octref.vetur",
    "editorconfig.editorconfig",
    "syler.sass-indented",

    "gruntfuggly.todo-tree"

  "unwantedRecommendations": []
}

```
